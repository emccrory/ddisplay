549  @Override
550  public void actionPerformed(ActionEvent e) {
551  int here = resultsTable.getSelectedRow();
552  int prev = here - 1;
553  int next = here + 1;
554  SelectedChannelsTableModel model = (SelectedChannelsTableModel) resultsTable.getModel();
555  switch (e.getActionCommand()) {
556  case "up":
557  if (model.swap(here, prev))
558  resultsTable.setRowSelectionInterval(prev, prev);
559  break;
560  case "down":
561  if (model.swap(here, next))
562  resultsTable.setRowSelectionInterval(next, next);
563  break;
564  case "delete":
565  model.delete(here);
566  if (model.getAllChannels().size() == 0) {
567  enableGroupButtons(false);
568  }
569  break;
570 
571  case "clear":
572  model.clear();
573  enableGroupButtons(false);
574  break;
575 
576  case "edit":
577  model.edit(here, this);
578  lengthOfThisListLabel.setText("Duration of list: " + resultsTable.getTotalTime());
579  break;
580 
581  default:
582  println(getClass(), "Invalid action command received: [" + e.getActionCommand() + "] This should not happen.");
583  break;
