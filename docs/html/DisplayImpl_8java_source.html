129  informListeners(DisplayChangeEvent.DisplayChangeType.CHANGE_RECEIVED, null);

131  channel = c;

133  if (localSetContent())
134  ;
135 
136  return channel;
137  }
138  boolean verified = isVerifiedChannel(c);
139  if (verified) {

143  println(getClass(), " -- Current channel is an EmergencyCommunication. Retaining the old previous channel of "
144  + previousChannelStack.peek());
145  else
146  previousChannelStack.push(channel);
147  }
148  if (c == null)
149  channel = makeEmptyChannel(null);
150  else
151  channel = c;

154  // informListeners(DisplayChangeEvent.Type.CHANGE_RECEIVED, null);
155 
156  // Actually do the channel change!
157  respondToContentChange(localSetContent(), "");
158  return previousChannelStack.peek();
159  } else {
160  error("The requested channel (URL=" + c.getURI().toString() + ") is not approved to be shown");
161  informListeners(DisplayChangeEvent.DisplayChangeType.ERROR,
162  "The requested channel (URL=" + c.getURI().toString() + ") is not approved to be shown");
163  }
