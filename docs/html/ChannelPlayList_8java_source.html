181  public long getChecksum() {
182  // I bet there is a more efficient way than this!
183 
184  // The string, ["http://1http://2"] WILL NOT return the same checksum as ["http://2http://1"] (verified)
185 
186  Checksum checksum = new CRC32();
187 
188  List<byte[]> allBytes = new ArrayList<byte[]>();
189  int totalLength = 0;
190  for (SignageContent C : channels) {
191  byte b[] = C.getURI().toASCIIString().getBytes();
192  totalLength += b.length;
193  allBytes.add(b);
194  }
195  byte ab[] = new byte[totalLength];
196  int index = 0;
197  for (byte[] B : allBytes) {
198  for (int k = 0; k < B.length; k++)
199  ab[index + k] = B[k];
200  index += B.length;
201  }
202  // update the current checksum with the specified array of bytes
203  checksum.update(ab, 0, ab.length);
204 
205  // get the current checksum value
206  return checksum.getValue();
207  }
