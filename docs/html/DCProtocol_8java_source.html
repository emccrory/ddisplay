305  private void createChannelListAndInform(ChangeChannelList mess) {
306  final ChannelSpec[] specs = mess.getChannelSpec();
307  if (specs.length == 0) {
308  println(DCProtocol.class, " -- Empty list received. Abort.");
309  return;
310  }
311  if (specs.length == 1) {
312  // There is only one channel in this list. Hmmm (shouldn't happen). Don't send this over and over to the client. The
313  // timeout in the channel itself will cause it to refresh from within the client.
314  informListeners(specs[0]);
315  } else {
316  long dwellTime = 5000;
317  ChannelListHolder channelList = new ConcreteChannelListHolder();
318 
319  for (ChannelSpec spec : specs) {
320  // NOTE : Lists of lists are not supported
321  ChannelInList c = new ChannelInListImpl(spec.getContent());
322  channelList.channelAdd(c);
323  }
324  ChannelPlayList playList = new ChannelPlayList( channelList, dwellTime);
325  playList.setChannelNumber(mess.getChannelNumber());
326  for (Display L : listeners) {
327  L.setContent(playList);
328  }
329  }
330  }
