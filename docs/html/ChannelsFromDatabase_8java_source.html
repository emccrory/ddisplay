248  private static final String decode(final String in) {

250  int index;
251  index = working.indexOf("\\u");
252  while (index > -1) {
253  int length = working.length();
254  if (index > (length - 6))
255  break;
256  int numStart = index + 2;
257  int numFinish = numStart + 4;
258  String substring = working.substring(numStart, numFinish);
