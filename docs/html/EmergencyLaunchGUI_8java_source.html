515  protected static String interp(long val) {
516  DecimalFormat format = new DecimalFormat("#.0");
517  String t = "" + val;
518  if (val <= 100L) {
519  return " seconds ( testing only )";
520  } else if (val < 3600) {
521  double min = val / 60.0;
522  t = format.format(min) + " minute" + (min != 1 ? "s" : "");
523  } else {
524  double hours = ((double) val) / (60 * 60);
525  t = format.format(hours) + " hour" + (hours != 1 ? "s" : "");
526  }
527  return " seconds ( " + t + " )";
528  }
