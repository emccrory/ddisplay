114  private void setTimeToWaitForResponses(boolean didTheyAllRespond) {
115  // The default that seems to work most of the time is 2 seconds. But occasionally, this gets delayed, so we have to lengthen
116  // this time. I made this method so I could try to be smart about it. If we are always getting all of them to respond, the
117  // we can reduce the wait time. If we do not see all of the responses, then we should increase the wait time until we DO see
118  // them all.
119 
120  // On my Linux test server, wait times as low as 1300 ms work well, and 1900 works virtually all the time
121 
122  // On a Windows PC, we're seeing that 2500 works most of the time.
123 
124  if (didTheyAllRespond) {
125  timeToWaitForResponses -= WAIT_TIME_INCREMENT;
126 
127  if (timeToWaitForResponses < minimumWaitTime)
128  timeToWaitForResponses = minimumWaitTime;
129  if (timeToWaitForResponses == minimumWaitTime) {
130  if (numTriesAtMinimum++ >= PATIENCE_VALUE) {
131  numTriesAtMinimum = 0;
132  // We sat at this wait time for "PATIENCE_VALUE" cycles, so let's decrement the minimum wait time a bit.
133  minimumWaitTime -= WAIT_TIME_INCREMENT;
134  }
135  }
136  } else {
137  // The last wait time was too long; reset the minimum.
138  minimumWaitTime = timeToWaitForResponses + WAIT_TIME_INCREMENT;
139  numTriesAtMinimum = 0;
140  timeToWaitForResponses += (PATIENCE_VALUE / 4) * WAIT_TIME_INCREMENT;
141  }
142 
143  // This is relevant since we often have a controller that cannot actually see all of its displays (that is, one of the
144  // displays is powered down, or the network is screwed up). It would be nice to actually KNOW if a display is down so the
145  // wait times can be set more intelligently. But, alas, ...
146  if (timeToWaitForResponses > ABSOLUTE_MAX_WAIT_TIME)
147  timeToWaitForResponses = ABSOLUTE_MAX_WAIT_TIME;
148 
149  if (debug)
150  println(getClass(), "timeToWaitForResponses = " + timeToWaitForResponses);
151  }
