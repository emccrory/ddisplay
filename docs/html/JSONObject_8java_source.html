208  public JSONObject(JSONTokener x) throws JSONException {
209  this();
210  char c;
211  String key;
212 
213  if (x.nextClean() != '{') {
214  throw x.syntaxError("A JSONObject text must begin with '{'");
215  }
216  for (;;) {
217  c = x.nextClean();
218  switch (c) {
219  case 0:
220  throw x.syntaxError("A JSONObject text must end with '}'");
221  case '}':
222  return;
223  default:
224  x.back();
225  key = x.nextValue().toString();
226  }
227 
228  // The key is followed by ':'.
229 
230  c = x.nextClean();
231  if (c != ':') {
232  throw x.syntaxError("Expected a ':' after a key");
233  }
234 
235  // Use syntaxError(..) to include error location
236 
237  if (key != null) {
238  // Check if key exists
239  if (this.opt(key) != null) {
240  // key already exists
241  throw x.syntaxError("Duplicate key \"" + key + "\"");
242  }
243  // Only add value if non-null
244  Object value = x.nextValue();
245  if (value!=null) {
246  this.put(key, value);
247  }
248  }
249 
250  // Pairs are separated by ','.
251 
252  switch (x.nextClean()) {
253  case ';':
254  case ',':
255  if (x.nextClean() == '}') {
256  return;
257  }
258  x.back();
259  break;
260  case '}':
261  return;
262  default:
263  throw x.syntaxError("Expected a ',' or '}'");
264  }

370  public JSONObject(String baseName, Locale locale) throws JSONException {
371  this();
372  ResourceBundle bundle = ResourceBundle.getBundle(baseName, locale,
373  Thread.currentThread().getContextClassLoader());
374 
375 // Iterate through the keys in the bundle.
376 
377  Enumeration<String> keys = bundle.getKeys();
378  while (keys.hasMoreElements()) {
379  Object key = keys.nextElement();
380  if (key != null) {
381 
382 // Go through the path, ensuring that there is a nested JSONObject for each
383 // segment except the last. Add the value using the last segment's name into
384 // the deepest nested JSONObject.
385 
386  String[] path = ((String) key).split("\\.");
387  int last = path.length - 1;
388  JSONObject target = this;
389  for (int i = 0; i < last; i += 1) {
390  String segment = path[i];
391  JSONObject nextTarget = target.optJSONObject(segment);
392  if (nextTarget == null) {
393  nextTarget = new JSONObject();
394  target.put(segment, nextTarget);
395  }
396  target = nextTarget;
397  }
398  target.put(path[last], bundle.getString((String) key));
399  }
