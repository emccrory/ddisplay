426  public static long getEntropy() {
427  // There are probably better ways to get a high-entropy value, but this one should be sufficient
428 
429  if (_myIPNumber == 0) {
430  try {
431  Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
432  while (interfaces.hasMoreElements()) {
433  NetworkInterface iface = interfaces.nextElement();
434  // filters out 127.0.0.1 and inactive interfaces
435  if (iface.isLoopback() || !iface.isUp())
436  continue;
437 
438  Enumeration<InetAddress> addresses = iface.getInetAddresses();
439  while (addresses.hasMoreElements()) {
440  InetAddress addr = addresses.nextElement();
441  String ip = addr.getHostAddress();
442  if (ip.contains(":"))
443  continue;
444  // System.out.print("--> " + ip);
445  String reverse = "";
446  for (int i = ip.length() - 1; i >= 0; i--) {
447  if (ip.charAt(i) == '.')
448  reverse += "1";
449  else
450  reverse += ip.charAt(i);
451  }
452  // System.out.println(" .. " + reverse + " ... " + System.nanoTime());
453 
454  // This is apparently not necessary as the Random class does an excellent job of giving a
455  // good random number sequence even with no seed. Whatever.
456  _myIPNumber = Long.parseLong(reverse);
457  SecureRandom sr = new SecureRandom();
458  byte[] ran = sr.generateSeed(6);
459  _goodSeed = _myIPNumber;
460  for ( int i=0; i<6; i++){
461  long b = 0x00ff & ran[i];
462  long c = (b<<(i*8));
463  _goodSeed = (c) ^ _goodSeed;
464  }
465  break;
466  }
467  }
468  } catch (SocketException e) {
