105  public JSONArray(JSONTokener x) throws JSONException {
106  this();
107  if (x.nextClean() != '[') {
108  throw x.syntaxError("A JSONArray text must start with '['");
109  }
110 
111  char nextChar = x.nextClean();
112  if (nextChar == 0) {
113  // array is unclosed. No ']' found, instead EOF
114  throw x.syntaxError("Expected a ',' or ']'");
115  }
116  if (nextChar != ']') {
117  x.back();
118  for (;;) {
119  if (x.nextClean() == ',') {
120  x.back();
121  this.myArrayList.add(JSONObject.NULL);
122  } else {
123  x.back();
124  this.myArrayList.add(x.nextValue());
125  }
126  switch (x.nextClean()) {
127  case 0:
128  // array is unclosed. No ']' found, instead EOF
129  throw x.syntaxError("Expected a ',' or ']'");
130  case ',':
131  nextChar = x.nextClean();
132  if (nextChar == 0) {
133  // array is unclosed. No ']' found, instead EOF
134  throw x.syntaxError("Expected a ',' or ']'");
135  }
136  if (nextChar == ']') {
137  return;
138  }
139  x.back();
140  break;
141  case ']':
142  return;
143  default:
144  throw x.syntaxError("Expected a ',' or ']'");
145  }
146  }
147  }
148  }
