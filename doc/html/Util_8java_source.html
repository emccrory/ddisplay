581  public static long getEntropy() {
582  // There are probably better ways to get a high-entropy value, but this one should be sufficient
583 
584  if (_myIPNumber == 0) {
585  try {
586  Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
587  while (interfaces.hasMoreElements()) {
588  NetworkInterface iface = interfaces.nextElement();
589  // filters out 127.0.0.1 and inactive interfaces
590  if (iface.isLoopback() || !iface.isUp())
591  continue;
592 
593  Enumeration<InetAddress> addresses = iface.getInetAddresses();
594  while (addresses.hasMoreElements()) {
595  InetAddress addr = addresses.nextElement();
596  String ip = addr.getHostAddress();
597  if (ip.contains(":"))
598  continue;
599  // System.out.print("--> " + ip);
600  String reverse = "";
601  for (int i = ip.length() - 1; i >= 0; i--) {
602  if (ip.charAt(i) == '.')
603  reverse += "1";
604  else
605  reverse += ip.charAt(i);
606  }
607  // System.out.println(" .. " + reverse + " ... " + System.nanoTime());
608  _myIPNumber = Long.parseLong(reverse);
609  break;
610  }
611  }
612  } catch (SocketException e) {
613  throw new RuntimeException(e);
614  }
615  _entropy = nextPrime(System.nanoTime() ^ _myIPNumber);
616  }
617  return _entropy;
618  }
